#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    float N1, N2;
    char operador;

    cout << "DIGITE UM NUMERO QUALQUER: " << endl;
    cin >> N1;

    cout << "DIGITE UM OPERADOR: " << endl;
    cout << "/: DIVISÃO" << endl;
    cout << "+: ADIÇÃO" << endl;
    cout << "-: SUBTRAÇÃO" << endl;
    cout << "*: MULTIPLICAÇÃO" << endl;
    cout << "^: ELEVAR" << endl;
    cin >> operador;

    cout << "DIGITE UM NUMERO QUALQUER: " << endl;
    cin >> N2;

    if (operador == '/') {
        cout << N1 / N2 << endl;
    }

    else if (operador == '+') {
        cout << N1 + N2 << endl;
    }

    else if (operador == '-') {
        cout << N1 - N2 << endl;
    }

    else if (operador == '*'){
        cout << N1 * N2 << endl;
    }

    else if (operador == '^') {
        cout << pow(N1,N2) << endl;
    }

    else {
        cout << "ERRO" << endl;
    }




    return 0;
}
